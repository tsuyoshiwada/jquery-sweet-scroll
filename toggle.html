HTML

<h2>Anchor Change state</h2>
<ul class="list-inline">
  <li><button type="button" id="btn-enable" class="btn btn-primary">Enable</button></li>
  <li><button type="button" id="btn-disable" class="btn btn-primary">Disable</button></li>
</ul>

<h2>Global Change state</h2>
<ul class="list-inline">
  <li><button type="button" id="btn-global-enable" class="btn btn-primary">Enable</button></li>
  <li><button type="button" id="btn-global-disable" class="btn btn-primary">Disable</button></li>
</ul>

JavaScript

$(function(){
  $("button").sweetScroll();

  var $anchor = $("a[href^='#']").sweetScroll();

  // Enable SweetScroll
  $("#btn-enable").on("click", function(){
    $anchor.sweetScroll("enable");
  });

  // Disable SweetScroll
  $("#btn-disable").on("click", function(){
    $anchor.sweetScroll("disable");
  });

  // Global enable SweetScroll
  $("#btn-global-enable").on("click", function(){
    $.sweetScroll("enable");
  });

  // Global disable SweetScroll
  $("#btn-global-disable").on("click", function(){
    $.sweetScroll("disable");
  });
});
