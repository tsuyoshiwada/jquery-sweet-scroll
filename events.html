HTML

<!-- Show status -->
<div id="status">
  Callback: <span id="status-callback"></span><br />
  Event: <span id="status-event"></span>
</div>

<ul class="list-inline">
  <li><a href="#block1" class="return-true btn btn-primary">void</a></li>
  <li><a href="#block2" class="return-false btn btn-primary">return false</a></li>
</ul>

JavaScript

$(function(){

  // Show status
  var $statusCallback = $("#status-callback"),
      $statusEvent = $("#status-event");

  // Callbacks & Events.
  $("a[href^='#']")
    .sweetScroll({
      beforeScroll: function(scrollProperties){
        if( $(this).hasClass("return-false") ){
          return false;
        }
        $statusCallback.text("beforeScroll");
      },
      afterScroll: function(scrollProperties){
        $statusCallback.text("afterScroll");
      },
      cancelScroll: function(){
        $statusCallback.text("cancelScroll");
      }
    })
    .on("sweetscroll.beforeScroll", function(e){
      $statusEvent.text("sweetscroll.beforeScroll");
    })
    .on("sweetscroll.afterScroll", function(e){
      $statusEvent.text("sweetscroll.afterScroll");
    })
    .on("sweetscroll.cancelScroll", function(e){
      $statusEvent.text("sweetscroll.cancelScroll");
    });

});
