Dynamically added elements

SweetScroll is valid even for dynamically added elements.

Try Example

    HTML

    <p><button type="button" id="btn-add" class="btn btn-primary">Add Link!</button></p>
<ul id="holder" class="list-inline"></ul>

<div id="block-holder"></div>

    JavaScript

    $(function(){
  $("a[href^='#']").sweetScroll();

  var $holder = $("#holder"),
      $blockHolder = $("#block-holder"),
      count = 0;

  $("#btn-add").on("click", function(){
    count++;

    var id = "block" + count;
    $holder.append('<li><a class="btn btn-primary" href="#' + id + '">Block' + count + '</a></li>');
    $blockHolder.append('<div class="block" id="' + id + '"><h2>Block' + count + '</h2></div>');
  });
});
