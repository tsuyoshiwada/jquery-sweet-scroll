Vertical & Horizontal

You can also be used to scroll horizontally to other than vertical.
To use this feature you will need to enable the horizontalScroll option.

Go to

HTML

<div class="form-inline">
  <div class="form-group">
    <select id="value-target" class="form-control input-lg">
      <option value="#block1">#block1</option>
      <option value="#block2">#block2</option>
      <option value="#block3">#block3</option>
      <option value="#block4">#block4</option>
      <option value="#block5">#block5</option>
      <option value="#block6">#block6</option>
      <option value="#block7">#block7</option>
      <option value="#block8">#block8</option>
      <option value="#block9">#block9</option>
      <option value="#block10">#block10</option>
    </select>
  </div>
  <div class="form-group">
    <button type="button" id="btn-scroll" class="btn btn-lg btn-primary">Go!</button>
  </div>
</div>

JavaScript

$(function(){

  $("#btn-scroll").on("click", function(){
    $.sweetScroll({
      target: $("#value-target").val(),
      verticalScroll: true,
      horizontalScroll: true
    });
  });

});
